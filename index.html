ระบบตรวจเช็กและสร้างตารางอัตโนมัติ (Auto-Generator)

โค้ด Apps Script ด้านล่างนี้ได้รับการพัฒนาให้**ตรวจเช็กและสร้างชีตย่อยและหัวข้อที่จำเป็นทั้งหมดในไฟล์ Google Sheet ให้คุณโดยอัตโนมัติ** คุณไม่จำเป็นต้องสร้างหัวตารางเองเลย!

ขั้นตอนการใช้โค้ดสร้างตารางอัตโนมัติ:

  1. สร้างไฟล์ Google Sheets เปล่าขึ้นมา 1 ไฟล์
  2. กดที่เมนู ส่วนขยาย (Extensions) > Apps Script
  3. ลบโค้ดเดิมในนั้นออกทั้งหมด แล้วคัดลอกโค้ดด้านล่างนี้ไปวาง
  4. กดไอคอนบันทึก (แผ่นดิสก์) จากนั้นกดปุ่ม ทำให้ใช้งานได้ (Deploy) > การทำให้ใช้งานได้ใหม่ (New deployment)
  5. เลือกประเภทเป็น เว็บแอป (Web app)
  6. ตั้งค่าผู้มีสิทธิ์เข้าถึงเป็น ทุกคน (Anyone) และกด Deploy ยอมรับสิทธิ์การเข้าถึง
  7. คัดลอก URL เว็บแอป ที่ได้มาวางในช่องด้านบน แล้วกด "บันทึกตั้งค่า" เป็นอันเสร็จสิ้น!
Apps Script (Auto-Create) 
function doGet(e) {
  var activeSpreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = activeSpreadsheet.getSheetByName("Students");
  
  // สร้างชีต "Students" อัตโนมัติถ้ายังไม่มีในไฟล์
  if (!sheet) {
    sheet = activeSpreadsheet.insertSheet("Students");
    sheet.appendRow(["roll", "name"]);
  }
  
  var data = sheet.getDataRange().getValues();
  var result = [];
  if (data.length > 1) {
    var headers = data[0];
    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      var obj = {};
      for (var j = 0; j < headers.length; j++) {
        obj[headers[j]] = row[j];
      }
      result.push(obj);
    }
  }
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var params = JSON.parse(e.postData.contents);
  var activeSpreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  
  if (params.action === "syncStudents") {
    var studentSheet = activeSpreadsheet.getSheetByName("Students");
    if (!studentSheet) {
      studentSheet = activeSpreadsheet.insertSheet("Students");
    } else {
      studentSheet.clearContents();
    }
    
    // เขียนหัวตารางและใส่ข้อมูลอัตโนมัติ
    studentSheet.appendRow(["roll", "name"]);
    params.students.forEach(function(student) {
      studentSheet.appendRow([student.roll, student.name]);
    });
    
    return ContentService.createTextOutput(JSON.stringify({status: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  if (params.action === "saveAttendance") {
    var attSheet = activeSpreadsheet.getSheetByName("AttendanceHistory");
    // สร้างชีต "AttendanceHistory" อัตโนมัติถ้ายังไม่มีในไฟล์
    if (!attSheet) {
      attSheet = activeSpreadsheet.insertSheet("AttendanceHistory");
      attSheet.appendRow(["Date", "Roll", "Name", "Status"]);
    }
    
    params.records.forEach(function(rec) {
      attSheet.appendRow([params.date, rec.roll, rec.name, rec.status]);
    });
    
    return ContentService.createTextOutput(JSON.stringify({status: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
